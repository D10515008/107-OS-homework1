項目二:

OS的工作為共享電腦資源給各程式/管理與提供介面讓程式不受硬體差異進行操作/共用資料，process有各自空間(指令/資料/stack)，呼叫系統功能使用system call

OS的功能:

  • IO/File Descriptor:
    用於讀寫外部輸入，具抽象化功能讓不同裝置的內容以byte串呈現
  • Pipe:
    提供方法讓process互相溝通
  • File system:
    管理檔案與資料(可以做link產生數個接口)，樹狀結構

OS需要能支援多工，分配時間給所有處理程序，並不讓它們互相影響。Xv6多數低階功能是x86專有的。

核心類型:

  • 單核心:
    應用程式直接使用核心的功能，效能高，但開發一但出錯會整個掛掉
  • 微核心:
    減少單核心的風險，讓核心功能模組化並獨立於系統，透過其他通訊機制來做操作

電腦開機初始化自己後從硬碟載入boot loader到主記憶體並執行。而boot loader將xv6的核心載入0x100000並從entry開始執行(此時x86的paging硬體還沒啟動，虛擬地址直接指向實體地址)，entry開始對各個虛擬地址對應實體地址。之後載入C程式，建立stack指標%esp，然後跳到main。接著main初始化裝置與子系統後呼叫userinit(叫allocproc在process table分配一個空間並初始化其狀態，之後分配核心stack放該process的執行緒)，核心執行緒開始執行暫存器內容(來自p->context)。userinit呼叫setupkvm建立page table(僅對應核心要用的記憶體)，process初始化後userinit會將其標記為可執行讓排程能用。main執行完userinit後mpmain讓排程器開始執行process(尋找可執行的process，此時只有initproc，呼叫switchuvm讓硬體使用該process的page table)，allocproc將initproc的p->context->eip設給forkret讓ret開始執行，之後讓stack頂部p->context之後的自備pop給trapret接著執行。Trapret用pop從trap frame復原暫存器，最後iret指令pop %cs，%eip，%flag，%esp，%ss，CPU繼續trap frame的%eip。initcode.S觸發system call exec(取代記憶體與暫存器當前的process，但file descriptor/process id/父process不變)，並將argv/$init/$0推到stack並要求核心執行exec。

項目三:

Page table:

讓OS在執行時知道各記憶體對應位置的機制，一個page為4096byte，以兩層page table方式管理，先用前10bit尋找page table，再用下10bit的虛擬地址找PTE

  • PTE狀態
    PTE_P:存在
    PTE_W:可寫
    PTE_U:使用者
    PTE_PWT:寫過
    PTE_PCD:不能用cache
    PTE_A:可存取
    PTE_D:髒的
    PTE_PS:大小
    PTE_MBZ:位元需是0

Process address space:

main為了另外管理空間呼叫kvmalloc切換到新的page table，每個process有各自page table(隨xv6切換process切換page table)。memlayout.h宣告了記憶體內容的常數以及將虛擬地址轉為實體地址的巨集。當process需要更多空間xv6先找閒置實體page後加入PTE到process的page table並設flag。xv6確保每個process有獨立記憶體(各自page table)且都從虛擬地址0開始。

Creating an address space:

main呼叫kvmalloc建立並切換到page table對應到KERNBASE之上讓kernel執行，多數工作由setupkvm執行








//分配一個page table給核心地址空間讓排程器process
voidkvmalloc(void)
{
    kpgdir = setupkvm();//此功能主要工作
    switchkvm();//切換到核心page table
}





















pde_t*setupkvm(void)
{
    pde_t *pgdir;
    struct kmap *k;

    if((pgdir = (pde_t*)kalloc()) == 0)
        return 0;
    memset(pgdir, 0, PGSIZE);//建立page大小記憶體
    if (P2V(PHYSTOP) > (void*)DEVSPACE)
        panic("PHYSTOP too high");
    for(k = kmap; k <&kmap[NELEM(kmap)]; k++)
        if(mappages(pgdir,k−>virt,k−>phys_end−k−>phys_start,uint)k−>phys_start, k−>perm) < 0)//安裝核心需要的轉換(核心指令/資料)
        {
            freevm(pgdir);
            return 0;
        }
    return pgdir;
}















static struct kmap {
    void *virt;
    uint phys_start;
    uint phys_end;
    int perm;
} 
kmap[] = {
{ (void*)KERNBASE, 0, EXTMEM, PTE_W}, // I/O空間
{ (void*)KERNLINK, V2P(KERNLINK), V2P(data), 0}, // kern text+rodata
{ (void*)data, V2P(data), PHYSTOP, PTE_W}, // kern data+memory
{ (void*)DEVSPACE, DEVSPACE, 0, PTE_W}, // more devices
};
























static int mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm)
{
    char *a, *last;
    pte_t *pte;

    a = (char*)PGROUNDDOWN((uint)va);
    last = (char*)PGROUNDDOWN(((uint)va) + size − 1);
    for(;;)//尋找虛擬地址要對應的PTE並獲得資料
    {
        if((pte = walkpgdir(pgdir, a, 1)) == 0)
            return −1;
        if(*pte & PTE_P)
            panic("remap");
        *pte = pa | perm | PTE_P;//初始化PTE以保留頁碼/所需權限/紀錄成有效
        if(a == last)
            break;
        a += PGSIZE;
        pa += PGSIZE;
    }
    return 0;
}






















static pte_t *walkpgdir(pde_t *pgdir, const void *va, int alloc)
{
    pde_t *pde;
    pte_t *pgtab;

    pde = &pgdir[PDX(va)];//使用虛擬地址前10bit找page目錄進入點
    if(*pde & PTE_P)//若不存在表示尚未被分配
    {
        pgtab = (pte_t*)P2V(PTE_ADDR(*pde));
    } 
    else 
    {
        if(!alloc || (pgtab = (pte_t*)kalloc()) == 0)
            return 0;
        memset(pgtab, 0, PGSIZE);
        *pde = V2P(pgtab) | PTE_P | PTE_W | PTE_U;
    }
    return &pgtab[PTX(va)];//使用後10bit在page table page找PTE地址
}

Physical memory allocation:

核心在執行期間需要分配與釋放記憶體給page table/process user memory/kernel stack/pipe buffer，每次都是4096byte。xv6在entry用分離的page分配器(不支援釋放且上限為4MB)

Physical memory allocator:

Allocator的資料結後是個實體記憶體page的free list，每個空page的清單元素都是run的結構




struct run {  
    struct run *next;//link list
};

Free list由spin lock保護







struct {
    struct spinlock lock;
    int use_lock;
    struct run *freelist;
} kmem;

main透過kinit1和kinit2初始化allocator












void kinit1(void *vstart, void *vend)//用前4MB建立沒被鎖的分配 
{
    initlock(&kmem.lock, "kmem");
    kmem.use_lock = 0;
    freerange(vstart, vend);//透過每個page呼叫kfree增加記憶體到free list
}
void kinit2(void *vstart, void *vend)//使用locking並安排更多可分配的記憶體
{
    freerange(vstart, vend); //透過每個page呼叫kfree增加記憶體到free list
    kmem.use_lock = 1;
}








freerange(void *vstart, void *vend)
{
    char *p;
    p = (char*)PGROUNDUP((uint)vstart);//確保只釋放對其的實體地址
    for(; p + PGSIZE <= (char*)vend; p += PGSIZE)
        kfree(p);//allocator起始沒有記憶體，用來管理
}

將記憶體中所有byte設為1，讓程式在記憶體被釋放後使用時是讀垃圾而非舊資料















void kfree(char *v) 
{ 
    struct run *r;
    if((uint)v % PGSIZE || v < end || V2P(v) >= PHYSTOP)
    panic("kfree");
    memset(v, 1, PGSIZE);//所有值都設為1
    if(kmem.use_lock)
        acquire(&kmem.lock);
    r = (struct run*)v; 
    r−>next = kmem.freelist; //紀錄舊的free list的start給r->next
    kmem.freelist = r;
    if(kmem.use_lock)
        release(&kmem.lock);
}

Sbrk:

Process需要增加或減少其記憶體時所用的system call，由growproc實作，此功能會更動process的page table(透過allocuvm與deallocuvm更新)，x86硬體將page table entry快取到TLB且當xv6改變page table就必須使被快取的entry無效(透過重新載入cr3暫存器的地址)




















int growproc(int n)
{
    uint sz;
    struct proc *curproc = myproc();
    sz = curproc−>sz;
    if(n > 0)//增加更多page對應到該process的頂部
    {
        if((sz = allocuvm(curproc−>pgdir, sz, sz + n)) == 0)//核心更新table
            return −1;
    } 
    else if(n < 0)//解除該process的地址空間對應並釋放對應的實體記憶體
    {
        if((sz = deallocuvm(curproc−>pgdir, sz, sz + n)) == 0) //核心更新table
            return −1;
    } 
    curproc−>sz = sz;
    switchuvm(curproc);
    return 0;
}

Exec:

建立使用者地址空間的system call，從檔案系統中的檔案初始化使用者部分的地址空間。exec從ELF檔案載入記憶體，使用者或process能放任何地址給ELF檔案，因此有風險(地址可指向核心，導致崩潰或顛覆隔離機制)，xv6執行很多檢查避免此問題(像是否overflow)。使用者能用ph.vaddr建立二進位ELF指向核心並讓ph.memsz大到能overflow到0x1000(此方法能通過allocuvm檢查)，之後loaduvm自動傳入ph.vaddr到核心(跳過檢查)從而複製二進位ELF資料到核心，如此使用者程式就能以核心權限執行，因此參數檢查很重要(xv6沒有完整的檢查所以隔離是能被繞過的)

































































































int exec(char *path, char **argv) 
{ 
    char *s, *last;
    int i, off;
    uint argc, sz, sp, ustack[3+MAXARG+1];
    struct elfhdr elf;
    struct inode *ip;
    struct proghdr ph;
    pde_t *pgdir, *oldpgdir;
    struct proc *curproc = myproc();
    begin_op();
    if((ip = namei(path)) == 0)//使用namei開啟被命名的二進位路徑
    {
        end_op();
        cprintf("exec: fail\n");
        return −1;
    }
    ilock(ip);
    pgdir = 0; 
    // 檢查ELF標頭
    if(readi(ip, (char*)&elf, 0, sizeof(elf)) != sizeof(elf))//讀取ELF標頭
        goto bad;
    if(elf.magic != ELF_MAGIC)//檢查二進位格式是否正確
        goto bad;
    if((pgdir = setupkvm()) == 0)//分配沒使用者對應的新page table
        goto bad;
    // 載入程式到記憶體
    sz = 0;
    for(i=0, off=elf.phoff; i<elf.phnum; i++, off+=sizeof(ph))
    {
        if(readi(ip, (char*)&ph, off, sizeof(ph)) != sizeof(ph))
            goto bad;
        if(ph.type != ELF_PROG_LOAD)
            continue;
        if(ph.memsz < ph.filesz)
            goto bad;
        if(ph.vaddr + ph.memsz < ph.vaddr)
            goto bad;
        if((sz = allocuvm(pgdir, sz, ph.vaddr + ph.memsz)) == 0)//為每個ELF區塊分配記憶體並檢查虛擬地址的要求在KERNBASE之下
            goto bad;
        if(ph.vaddr % PGSIZE != 0)
            goto bad;
        if(loaduvm(pgdir, (char*)ph.vaddr, ip, ph.off, ph.filesz) < 0)//將每個區塊載入記憶體
            goto bad;
    }
    iunlockput(ip);
    end_op();
    ip = 0;
    // 在下個page邊界分配兩個page
    // 讓第一個不可用，第二個當使用者stack
    sz = PGROUNDUP(sz);
    if((sz = allocuvm(pgdir, sz, sz + 2*PGSIZE)) == 0)
        goto bad;
    clearpteu(pgdir, (char*)(sz − 2*PGSIZE));
    sp = sz;
    // 推入字串參數
    for(argc = 0; argv[argc]; argc++) 
    {
        if(argc >= MAXARG)
            goto bad;
        sp = (sp − (strlen(argv[argc]) + 1)) & ~3;
        if(copyout(pgdir, sp, argv[argc], strlen(argv[argc]) + 1) < 0)//檢查目標page是否可用
            goto bad; 
        ustack[3+argc] = sp;
    } 
    ustack[3+argc] = 0;
    ustack[0] = 0xffffffff; // fake return PC
    ustack[1] = argc;
    ustack[2] = sp − (argc+1)*4; // argv 指標
    sp −= (3+argc+1) * 4;
    if(copyout(pgdir, sp, ustack, (3+argc+1)*4) < 0)//偵測是否為無效程式(若是則釋放新的image並回傳-1)
        goto bad;
    //存程式名用於debug
    for(last=s=path; *s; s++)
        if(*s == ’/’)
    last = s+1;
    safestrcpy(curproc−>name, last, sizeof(curproc−>name));
    // Commit to the user image.
    oldpgdir = curproc−>pgdir;
    curproc−>pgdir = pgdir;   
    curproc−>sz = sz; 
    curproc−>tf−>eip = elf.entry;  // main
    curproc−>tf−>esp = sp;
    switchuvm(curproc);//安裝新image
    freevm(oldpgdir);//釋放舊image
    return 0;
    bad:
        if(pgdir)
            freevm(pgdir);
        if(ip)
        {
            iunlockput(ip);
            end_op();
        }
    return −1;
}





















int loaduvm(pde_t *pgdir, char *addr, struct inode *ip, uint offset, uint sz)
{
    uint i, pa, n;
    pte_t *pte;
    if((uint) addr % PGSIZE != 0)
        panic("loaduvm: addr must be page aligned");
    for(i = 0; i < sz; i += PGSIZE)
    {
        if((pte = walkpgdir(pgdir, addr+i, 0)) == 0)//尋找被分配記憶體的實體地址，在內寫入每個page的ELF區塊
            panic("loaduvm: address should exist");
        pa = PTE_ADDR(*pte);
        if(sz − i < PGSIZE)
            n = sz − i;
        else
            n = PGSIZE;
        if(readi(ip, P2V(pa), offset+i, n) != n)//從檔案讀入
            return −1;
    }
    return 0;
}






















Int copyout(pde_t *pgdir, uint va, void *p, uint len){
    char *buf, *pa0;
    uint n, va0;
    buf = (char*)p;
    while(len > 0)
    {
        va0 = (uint)PGROUNDDOWN(va);
        pa0 = uva2ka(pgdir, (char*)va0);
        if(pa0 == 0)//檢查目標page是否可存取
            return −1;
        n = PGSIZE − (va − va0);
        if(n > len)
            n = len;
        memmove(pa0 + (va − va0), buf, n);
        len −= n;
        buf += n;
        va = va0 + PGSIZE;
    }
    return 0;
}




















struct elfhdr {
    uint magic;// must equal ELF_MAGIC//ELF開頭有4byte(0x7F/E/L/F)或ELF_MAGIC，有此資訊時表示其二進位資料格式正確
    uchar elf[12];
    ushort type;
    ushort machine;
    uint version;
    uint entry;
    uint phoff; 
    uint shoff;
    uint flags;
    ushort ehsize;
    ushort phentsize;
    ushort phnum;
    ushort shentsize;
    ushort shnum;
    ushort shstrndx;
};
