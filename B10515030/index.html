CH1 Operating System Organization

Process是一個抽象概念，讓他自己覺得獨佔一台機器，使得其他process無法讀寫。Xv6使用page table為每一個process提供獨有的空間，將虛擬位置(x86指令使用的地址)映射為實際位置(CPU對main memory溝通的地址)。

Xv6為每個process維護不同的page table，因此可以合理地定義process address，如圖1-1一個address space包含虛擬地址從0開始的user memory, global variable, stack 以及一個可供額外要求的記憶體空間heap(for malloc)。

Xv6使用$struct proc 維護一個process的狀態，特別重要的kernel state是page table, kernel stack and run state。每一個process都有一個thread來執行process的指令，thread可以被暫停之後再繼續，thread大部分的狀態(local variable and return address)都存在thread的stack上。

當process呼叫system call時，processor切換至kernel stack提升它的特權，然後執行system call的kernel instruction，執行完成後，從kernel space 回到 user space降低它的特權切換回user stack，繼續執行system call下一條指令。

$p->state : 顯示process的狀態，e.g.: process is allocated, ready to run, running, waiting, for I/O, or exiting.
$p->pgdir : 以x86硬體要求的格式儲存process的page table。

項目三

CH2 Page Tables

Xv6藉由page table讓process memory映射到實體記憶體，且防止自己的記憶體空間讓其他process存取。

結構

利用兩層的tree結構化實體記憶體，Root是一個4KB的page directory組成，其中包含1024個類PTE結構的page table pages，每個page table page 包含1024個32-bits的PTE。
PTE : Page Table Entry，32位元，包含20位元的physical page number(PPN)和12位元的旗標，旗標用於說明page hardware對virtual address

用virtual address的高位元10個bits從page directory選擇對應的page table，再從剩下的10bits依據page table選擇對應的PTE，最後將選到的PTE最高20位元作為physical address的最高20位元，physical address剩下的12位元則是virtual address的後12位元。

//Set up kernel part of page table
pde_t* setupkvm(void)
{
  pde_t *pgdir;
  struct kmap *k;

  if((pgdir = (pde_t*)kalloc()) == 0)
    return 0;
  memset(pgdir, 0, PGSIZE);
  if (P2V(PHYSTOP) > (void*)DEVSPACE)
    panic("PHYSTOP too high");
  for(k = kmap; k < &kmap[NELEM(kmap)]; k++)
    if(mappages(pgdir, k->virt, k->phys_end - k->phys_start,
                (uint)k->phys_start, k->perm) < 0) {
      freevm(pgdir);
      return 0;
    }
  return pgdir;
}





















static int mappages(pde_t *pgdir, void *va, uint size, uint pa, int perm)
{
  char *a, *last;
  pte_t *pte;

  a = (char*)PGROUNDDOWN((uint)va);
  last = (char*)PGROUNDDOWN(((uint)va) + size - 1);
  for(;;){
    if((pte = walkpgdir(pgdir, a, 1)) == 0)
      return -1;
    if(*pte & PTE_P) 
      panic("remap");
    *pte = pa | perm | PTE_P;
    if(a == last)
      break;
    a += PGSIZE;
    pa += PGSIZE;
  }
  return 0;
}





















static pte_t * walkpgdir(pde_t *pgdir, const void *va, int alloc)
{
  pde_t *pde;
  pte_t *pgtab;

  pde = &pgdir[PDX(va)];
  if(*pde & PTE_P){
    pgtab = (pte_t*)P2V(PTE_ADDR(*pde));
  } else {
    if(!alloc || (pgtab = (pte_t*)kalloc()) == 0)
      return 0;
    // Make sure all those PTE_P bits are zero.
    memset(pgtab, 0, PGSIZE);
    // The permissions here are overly generous, but they can
    // be further restricted by the permissions in the page table
    // entries, if necessary.
    *pde = V2P(pgtab) | PTE_P | PTE_W | PTE_U;
  }
  return &pgtab[PTX(va)];
}

// Grow current process’s memory by n bytes.
// Return 0 on success, −1 on failure.
int growproc(int n)
{
  uint sz;
  struct proc *curproc = myproc();
  sz = curproc−>sz;
  if(n > 0){
    //allocate physical page and maps them at the top of process's address space
    if((sz = allocuvm(curproc−>pgdir, sz, sz + n)) == 0)
      return −1;
    } else if(n < 0){
     //ummap pages from process's address space 然後free掉對應的physical pages
     if((sz = deallocuvm(curproc−>pgdir, sz, sz + n)) == 0)
    return −1;
  }
  curproc−>sz = sz;
  switchuvm(curproc);
  return 0;
}

exec

Exec is a system call which usually follow $fork system call, after $fork() create new child process, $exec() allocate it a different memory space for running.



















































































































int exec(char *path, char **argv)
{
  char *s, *last;
  int i, off;
  uint argc, sz, sp, ustack[3+MAXARG+1];
  struct elfhdr elf;
  struct inode *ip;
  struct proghdr ph;
  pde_t *pgdir, *oldpgdir;
  struct proc *curproc = myproc();
  
  begin_op();
  
  if((ip = namei(path)) == 0){//使用namei開啟binary path
    end_op();
    cprintf("exec: fail\n");
    return −1;
  }
  ilock(ip);
  pgdir = 0;
  
  // Check ELF header
  //檢查檔案是不是ELF格式，ELF的格式：開頭會有 4 bytes的magic number，
  //0X7F、E、L、F or ELF_MAGIC。
  if(readi(ip, (char*)&elf, 0, sizeof(elf)) != sizeof(elf))
    goto bad;
  if(elf.magic != ELF_MAGIC)
    goto bad;
    
  //allocate 一個沒有使用者對應的page
  if((pgdir = setupkvm()) == 0)
    goto bad;
  
  // Load program into memory.
  sz = 0;
  for(i=0, off=elf.phoff; i<elf.phnum; i++, off+=sizeof(ph)){
    if(readi(ip, (char*)&ph, off, sizeof(ph)) != sizeof(ph))
      goto bad;
    if(ph.type != ELF_PROG_LOAD)
      continue;
    if(ph.memsz < ph.filesz)
      goto bad;
    if(ph.vaddr + ph.memsz < ph.vaddr)
	goto bad;
    
    //allocuvm 會負責allocate page table 和 physical memory 給process
    if((sz = allocuvm(pgdir, sz, ph.vaddr + ph.memsz)) == 0)
      goto bad;
    if(ph.vaddr % PGSIZE != 0)
      goto bad;
      
    //loaduvm 將program 的ELF egment load 進pgdir中，而且address必須是page-aligned，
    //還有pages 的從addr到addr+sz都必須已經對應完成。
    if(loaduvm(pgdir, (char*)ph.vaddr, ip, ph.off, ph.filesz) < 0)
      goto bad;
    iunlockput(ip);
  }
  end_op();
  ip = 0;
  
  //exec allocate 和初始化 user stack
  // Allocate two pages at the next page boundary.
  // Make the first inaccessible. Use the second as the user stack.
  sz= PGROUNDUP(sz);
  if((sz = allocuvm(pgdir, sz, sz + 2*PGSIZE)) == 0)
    goto bad;
  clearpteu(pgdir, (char*)(sz − 2*PGSIZE));
  sp= sz;
	
  //Push argument strings, prepare rest of stack in ustack.
  for(argc = 0; argv[argc]; argc++)
  {
    if(argc >= MAXARG)
      goto bad;
    sp = (sp − (strlen(argv[argc]) + 1)) & ~3;
    if(copyout(pgdir, sp, argv[argc], strlen(argv[argc]) + 1) < 0)
      goto bad;
    ustack[3+argc] = sp;
  }
  ustack[3+argc] = 0;
  
  ustack[0] = 0xffffffff; // fake return PC
  ustack[1] = argc;
  ustack[2] = sp − (argc+1)*4; // argv pointer
  
  sp−= (3+argc+1) * 4;
  if(copyout(pgdir, sp, ustack, (3+argc+1)*4) < 0)
    goto bad;
	
  //Save program name for debugging.
  for(last=s=path; *s; s++)
    if(*s == ’/’)
      last = s+1;
  safestrcpy(curproc−>name, last, sizeof(curproc−>name));
  
  // Commit to the user image.
  oldpgdir = curproc−>pgdir;
  curproc−>pgdir = pgdir;
  curproc−>sz = sz;
  curproc−>tf−>eip = elf.entry; // main
  curproc−>tf−>esp = sp;
  switchuvm(curproc);
  freevm(oldpgdir);
  return 0;
  
  bad:
    if(pgdir)
      freevm(pgdir);
    if(ip){
      iunlockput(ip);
      end_op();
     }
  return −1;
}

項目二 :

Ch0

OS的工作有四個
1.將資源在多個program間共享，並提供比hardware更有用的服務
2.管理並抽象底層硬體
3.在多個program間共享硬體，使program同時運行(至少看起來像)
4.提供一種受控的交互方式，使program間可以共享數據、共同工作

OS透過interface提供服務，讓interface的設計依賴於少量的mechanism，通過這些mechanism的組合提供強大、通用的功能。

xv6使用kernel概念 - 一個向其他運行中program提供服務的特殊program。
每個運行中的program(稱作process)都擁有包含instrustions、data、stack的記憶體。

Process透過system call使用kernel服務。System call會進入kernel執行服務然後返回，所以process總是在user space和kernel space間交替運行。

kernel使用CPU的硬體保護機制來保證user的process只能access自己的記憶體。

Ch1

當PC開機時，會初始化自己並載入boot loader到memory並運行。
bootloader把xv6 kernel從disk中載入並從entry(1040)開始運行。
entry的code設置了page table，將0x80000000(KERNBASE)開始的虛擬地址map到物理地址0x0處。
Entry 512將虛擬地址KERNBASE:KERNBASE+0x400000 map到物理地址0:0x400000。這個mapping限制了kernel的指令+代碼必須再4mb以內。
entry會將entrypgdir的物理地址載入到暫存器 %cr3中。entrypgdir指向記憶體的高地址處，但我們可以用V2P_WO減掉KERNBASE就可以找到物理地址。
為了讓paging hardware運作，xv6會將flag CR0_PG設置在控制暫存器 %cr0中。
然後entry會跳轉到kernel的C code，並在memory的高地址中執行。
將stack pointer %esp指向被用作stack的一段memory(1054)之後，entry會jump到高地址的main代碼中。必須要使用indirect jump才不會造成PC-relative direct jump，而運行了在memory低地址處的main。

通過 userinit(1239)建立第一個process。
userinit先呼叫了 allocproc。allocproc會在page table中分配一個slot並初始化process state。
當allocproc在proc的表中找到標記為UNUSED的slot時，會將其狀態設置為EMBRYO，使其被標記為被使用的，並給予該process一個獨有的pid。接下來，會為了process的kernel thread分配kernel stack，如果失敗了，allocproc會將該slot的狀態恢復為UNUSED並return 0以標記失敗。
第一個process會先運行一個program(initcode.S)。
第一個process 的memory中的初始內容是由initcode.S編譯得到的，在build的過程中會產生兩個符號: _binary_initcode_start 和 _binary_initcode_size。userinit呼叫inituv，分配一頁物理記憶體，將虛擬地址0 map到那一段記憶體，並將code複製到那一頁中。
一旦process初始化完畢，userinit將 p->state設置為RUNNABLE。

在main呼叫userinit後，mpmain會呼叫scheduler進行process。
scheduler會找到一個 p->state為RUNNABLE 的process，並用 switchuvm使硬體使用目標process的page taable。
scheduler會將p->state改為RUNNING，呼叫swtch，保存當前的暫存器，並把目標thread的暫存器載入到x86的硬體暫存器。

initcode.S剛開始會將argv、argv、init、$0三個值push進stack中，然後把 %eax設置為SYS_exec並執行int T_SYSCALL。

項目三 :

Page table用二維陣列儲存。
Virtual address 的高10位用來找到page directory，接下來的10位用來選出該page table的PTE。
Physical address 的高20位是對應PTE的PPN。Virtual address的低12位是offset，page table hardware會原樣複製到Physical address的低12位。

// Free the page of physical memory pointed at by v,
// which normally should have been returned by a
// call to kalloc().  (The exception is when
// initializing the allocator; see kinit above.)
void
kfree(char *v)
{
  struct run *r;

  if((uint)v % PGSIZE || v < end || V2P(v) >= PHYSTOP)
    panic("kfree");

  // Fill with junk to catch dangling refs.
  memset(v, 1, PGSIZE);

  if(kmem.use_lock)
    acquire(&kmem.lock);
  r = (struct run*)v;
  r->next = kmem.freelist;
  kmem.freelist = r;
  if(kmem.use_lock)
    release(&kmem.lock);
}

// Allocate page tables and physical memory to grow process from oldsz to
// newsz, which need not be page aligned.  Returns new size or 0 on error.
int
allocuvm(pde_t *pgdir, uint oldsz, uint newsz)
{
  char *mem;
  uint a;

  if(newsz >= KERNBASE)
    return 0;
  if(newsz < oldsz)
    return oldsz;

  a = PGROUNDUP(oldsz);
  for(; a < newsz; a += PGSIZE){
    mem = kalloc();
    if(mem == 0){
      cprintf("allocuvm out of memory\n");
      deallocuvm(pgdir, newsz, oldsz);
      return 0;
    }
    memset(mem, 0, PGSIZE);
    if(mappages(pgdir, (char*)a, PGSIZE, V2P(mem), PTE_W|PTE_U) < 0){
      cprintf("allocuvm out of memory (2)\n");
      deallocuvm(pgdir, newsz, oldsz);
      kfree(mem);
      return 0;
    }
  }
  return newsz;
}

然後用 loaduvm把每個segment載入到記憶體中，並通過 walkpgdir來找到physical address。

// Load a program segment into pgdir.  addr must be page-aligned
// and the pages from addr to addr+sz must already be mapped.
int
loaduvm(pde_t *pgdir, char *addr, struct inode *ip, uint offset, uint sz)
{
  uint i, pa, n;
  pte_t *pte;

  if((uint) addr % PGSIZE != 0)
    panic("loaduvm: addr must be page aligned");
  for(i = 0; i < sz; i += PGSIZE){
    if((pte = walkpgdir(pgdir, addr+i, 0)) == 0)
      panic("loaduvm: address should exist");
    pa = PTE_ADDR(*pte);
    if(sz - i < PGSIZE)
      n = sz - i;
    else
      n = PGSIZE;
    if(readi(ip, P2V(pa), offset+i, n) != n)
      return -1;
  }
  return 0;
}
