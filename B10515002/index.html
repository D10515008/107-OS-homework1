Chapter4-Lock

Race condition

資源同時被多個CPU存取並修改其內容，可能導致某個CPU讀取的結果錯誤，此現象為 Race condition。
1 struct list {
2     int data;
3     struct list *next;
4 };
5
6 struct list *list = 0;
7
8 void insert(int data)//將檔案新增於當前list最前端
9 {
10     struct list *l;
11
12     l = malloc(sizeof *l);
13     l->data = data;
14     l->next = list;
15     list = l;
16 }

當兩個CPU同時執行 insert() 函式時，會在14,15行發生錯誤，兩個CPU同時將 l->next 指向當前的list，並在15行後，後執行的CPU會蓋掉前面執行CPU的結果。

Lock and Critical Section

Lock可以避免資源在多個CPU存取下，發生Race Condition的錯誤
被Lock起來的區塊稱為Critical Section，可以保證同一時間只有一個CPU執行此區塊中的程式碼，進而保持該資料不被同時改動，利用這個技巧可以保持不變性(invariant)，可以避免Race Condition的錯誤。
 1 struct list *list = 0;
 2 struct lock listlock;
 3
 4 void insert(int data)
 5 {
 6     struct list *l;
 7     acquire(&listlock);
 8 
 9     l = malloc(sizeof l);
 10    l->data = data;
 11    l->next = list;
 12    list = l;
 13    release(&listlock);
 14 }

可以將Critical Section 看成是序列化(Serialized)的程式碼區塊，較晚獲得Lock的CPU，看到的都是前一個CPU的執行結果。

Spin Lock

1. struct spinlock {
2.   uint locked; // 0代表lock可用，非0代表已被使用(held)
3. 
4.   // For debugging:
5.   char *name; // 此lock的名稱
6.   struct cpu *cpu; // 使用此lock的CPU
7.   uint pcs[10]; // 函式呼叫的stack，以PC(Program Counter)表示
8. };

Acquire Function

1 void
2 acquire(struct spinlock *lk)
3 {
4   for(;;) {
5     if(!lk->locked) {
6       lk->locked = 1;
7       break;
8     }
9   }
10}

上述Aquire本身有個Race Condition的問題在，可能兩個CPU同時存取第5行，導致2個CPU都獲得Lock，違反了本身aquire想達成的目的。

要避免上述情況發生，需使用原子操作，使用x86下特殊指令xchg，此指令用於將記憶體中的某個WORD與特定的值交換。

1  void
2  acquire(struct spinlock *lk)
3  {
4   pushcli(); // 為了避免deadlock，關閉中斷
5   if(holding(lk))
6        panic("acquire");
7 
8   // 使用xchg原子操作指令實現之前的功能(lk->lock = 1 xchg回傳1，若lk->lock = 0， xch與1交換，回傳0)
9   while(xchg(&lk−>locked, 1) != 0)
10  ;
11
12  // 保證編譯器不會重新安排指令執行順序
13  __sync_synchronize();
14
15  // set debugging information
16  lk−>cpu = mycpu();
17  getcallerpcs(&lk, lk−>pcs);
18 }

Release Function

要釋放lock，必須清除lock所保存的debug的資訊，並將locked設定回0，其中會利用原子操作，使用x86下的movl保證其正確性。
1 void
2 release(struct spinlock *lk)
3 {
4   if(!holding(lk))
5   panic("release");
6
7   lk−>pcs[0] = 0;
8   lk−>cpu = 0;
9
10   // 保證編譯器不會重新安排指令執行順序
11   __sync_synchronize();
12
13   // 使 lk−>locked = 0 的原子指令
14   asm volatile("movl $0, %0" : "+m" (lk−>locked) : );
15
16   //重新啟動中斷
17   popcli();
18 }

Sleep Lock

有些系統需要長時間獲得Lock,例如: File System，為了讓等待期間，其他Process也能持續進行，故有了 Sleep Lock。
1 struct sleeplock {
2   uint locked; // 是否被使用
3   struct spinlock lk; // 保護sleep lock
4
5   // For debugging:
6   char *name; // lock的名稱
7   int pid; // 使用此lock的process的id
8 };

Acquiresleep and Releasesleep Function

為了讓其他Process能繼續執行，acquiresleep會呼叫sleep，因此釋放自己的spin-lock，反之亦然。
1  void
2  acquiresleep(struct sleeplock *lk)
3  {
4    acquire(&lk−>lk);
5    while (lk−>locked) {
6      sleep(lk, &lk−>lk);
7    }
8    lk−>locked = 1;
9    lk−>pid = myproc()−>pid;
10   release(&lk−>lk);
11 }

1 void
2 releasesleep(struct sleeplock *lk)
3 {
4   acquire(&lk−>lk);
5   lk−>locked = 0;
6   lk−>pid = 0;
7   wakeup(lk);
8   release(&lk−>lk);
9 }

Limitation of Lock

某函式內有需要保護的資料，但他同時被2種函式呼叫，以取得Lock與未使用Lock的函式，此情況有兩種解決方法，讓呼叫者持有Lock或者是讓被呼叫者持有Lock。
